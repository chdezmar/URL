def delete(args={})

  resp = http.request(request)

def post(args={})
  http = http_obj
  request = Net::HTTP::Post.new(url.path)
  request.set_form_data(url.params)
  t = Time.now
  resp = http.request(request)
  make_str(resp,Time.now-t)
end
